IntelliJ Configurations Instructions
   100% Performance Configurations
         Appearance & Behavior
                  Navigate : File -> Settings -> Appearance & Behavior -> Appearance
                  Uncheck : "Animate windows"
                  Uncheck : "Use custom font"
                  Uncheck : "Show memory indicator"
                  Uncheck : "Enable HTML/SSR preview"
         Editor
                  Navigate : File -> Settings -> Editor -> General
                  Uncheck : "Show quick documentation on mouse move"
                  Uncheck : "Show parameter name hints"
                  Uncheck : "Show local variable type hints"
                  Uncheck : "Enable HTML/SSR preview"
                  Navigate : File -> Settings -> Editor -> Code Style
                  Set : "Line separator" to "System-dependent"
         Plugins
                  Navigate : File -> Settings -> Plugins
                  Disable : Unnecessary plugins (only keep Gradle, Maven, and testing-related plugins enabled)
         Build, Execution, Deployment
                  Navigate : File -> Settings -> Build, Execution, Deployment -> Compiler
                  Uncheck : "Clear output directory on rebuild"
                  Uncheck : "Automatically show first error in editor"
                  Uncheck : "Display notification on build completion"
                  Uncheck : "Build project automatically"
                  Uncheck : "Compile independent modules in parallel"
                  Uncheck : "Rebuild module on dependency change"
                  Input 767 Value : "Shared build process heap size (Mbytes)"
                  Navigate : File -> Settings -> Build, Execution, Deployment -> Gradle
                  Select : "Use Gradle from" -> "wrapper task in Gradle build script"
                  Check : "Offline work" (if you have all dependencies downloaded)
                  Set : "Build and run using" to "IntelliJ IDEA"
                  Set : "Run tests using" to "IntelliJ IDEA"
                  Navigate : File -> Settings -> Build, Execution, Deployment -> Maven
                  Set : "VM options for importer" -> "-Xmx2g -XX:ReservedCodeCacheSize=256m"
                  Check : "Use plugin registry"
                  Check : "Work offline" (if you have all dependencies downloaded)
         Languages & Frameworks
                  Navigate : File -> Settings -> Languages & Frameworks -> Groovy
                  Uncheck : "Enable Groovy++"
         Terminal
                  Navigate : File -> Settings -> Terminal
                  Uncheck : "Copy on selection"
                  Uncheck : "Paste on middle button click"
         Active Tool Window
                  Navigate : View -> Active Tool Window
                  Close : Close unnecessary tool windows to reduce visual clutter and improve performance
         Tasks & Contexts
                  Navigate : Tools -> Tasks & Contexts
                  Uncheck : "Enable issue cache"
                  Navigate : Window -> Store Current Layout as Default
                  Action : Customize and save your layout to reduce resource usage
         Store Current Layout as Default
                  Navigate : Window -> Store Current Layout as Default
                  Action : Customize and save your layout to reduce resource usage

         Help
                  Set : Edit Custom VM Options ->

                         -> -Xms2g
                            -Xmx4g
                            -XX:ReservedCodeCacheSize=512m
                            -XX:+UseConcMarkSweepGC
                            -XX:SoftRefLRUPolicyMSPerMB=50
                            -ea
                            -Dsun.io.useCanonCaches=false
                            -Djava.net.preferIPv4Stack=true
                            -Djdk.http.auth.tunneling.disabledSchemes=""
